I like procedural programming too, but this code would be much better:

procedure makesession;
var
  session: TEntity;
begin
  session := TEntity.Create('session');
  try
    session.New;
    // attributes are instantiated dynamically
    session.Att['value'] := cvalue;
    session.Att['currenttime'] := currenttime;
    session.Att['sessuser'] := sessuser;
    session.Save;
  finally
    session.Free;
  end;
end;
