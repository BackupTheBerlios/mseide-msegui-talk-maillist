Here is the code I changed in the main.pas source file.  Replace the 
tmainfo.initdata procedure with this.

procedure tmainfo.initdata(const sender: TObject);
begin
 treeedit.itemlist.add(tmynode.create); //root nodes
 with tmynode(treeedit[0]) do begin
  caption:= 'A0';
  str := 'A0';
  int := 1;
  
  add(1,tmynode);
  items[count -1].caption:= 'AA0';
  tmynode(items[count-1]).str := 'AA0';
  tmynode(items[count-1]).int := 0;
  
  add(1,tmynode);
  items[count -1].caption:= 'AA1';
  tmynode(items[count-1]).str := 'AA1';
  tmynode(items[count-1]).int := 1;

  with items [count-1]
  do begin
	add(1,tmynode);
	items[count-1].caption:= 'AA1a';
	tmynode(items[count-1]).str := 'AA1a';
	tmynode(items[count-1]).int := 1;
  end;

  add(1,tmynode);
  items[count -1].caption:= 'AA2';
  tmynode(items[count-1]).str := 'AA2';
  tmynode(items[count-1]).int := 2;
  
  add(1,tmynode);
  items[count -1].caption:= 'AA3';
  tmynode(items[count-1]).str := 'AA3';
  tmynode(items[count-1]).int := 3;
 end;
 
 // in a like manner you can do the same for the 2nd treeedit item
 treeedit.itemlist.add(tmynode.create); 

 with tmynode(treeedit[1]) do begin
 
  caption:= 'B0';
  str := 'B0';
  int := 1;
  
  add(1,tmynode);
  items[count -1].caption:= 'BB0';
  tmynode(items[count-1]).str := 'BB0';
  tmynode(items[count-1]).int := 0;
  
  add(1,tmynode);
  items[count -1].caption:= 'BB1';
  tmynode(items[count-1]).str := 'BB1';
  tmynode(items[count-1]).int := 1;

  add(1,tmynode);
  items[count -1].caption:= 'BB2';
  tmynode(items[count-1]).str := 'BB2';
  tmynode(items[count-1]).int := 2;
  
  add(1,tmynode);
  items[count -1].caption:= 'BB3';
  tmynode(items[count-1]).str := 'BB3';
  tmynode(items[count-1]).int := 3; 

  add(5,tmynode);
  items[4].caption:= 'BB0';
  items[5].caption:= 'B1';
  with tmynode(items[6]) do begin
   add(5,tmynode);
   items[0].caption:= 'B1a';
   items[1].caption:= 'B1b';
   items[2].caption:= 'B1c';

   items[4].caption:= 'B1e';

 end;

end;
