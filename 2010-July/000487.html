Try this code:


  // insert three rows
  tstringgrid1.datacols.count:= 1;
  tstringgrid1.rowcount:= 3;
  tstringgrid1[0][0] := 'one';
  tstringgrid1[0][1] := 'two';
  tstringgrid1[0][2] := 'three';

  // At this point you have three rows one, two, three


  // insert a row at position 1
  tstringgrid1.insertrow(1);
  // now you have one, empty, two, three
