It seems that because the font name is descended as parent then any change  
to the font is really a change to this .
So, the Q turns int :

how to change font size & style for fixrows created programically in  
run-time so that it won't affect its parent font ?

At a glance, "font:= tcolheaderfont.create;" for each  
fixrows[i],captrion*[j].font looks appropriate (and it really works see  
the below code), but won't it result  in non-freed heap memory ?



with grdForm18 do begin
    fixrows.count:= recordcount+1;
    fixrows.oppositecount:= recordcount;

    fixcols.count:= 1;
    for i:= 0 to fixrows.oppositecount-1 do begin
      with fixrows[-2-i] do begin
        color:= $F0FFF0; height:= 15;

        captionsfix.count:= 1;
        with captionsfix[-1] do begin
          font:= tcolheaderfont.create;
          font.height:= 12;
          font.style:= font.style + [fs_bold];
          captionsfix[-1].caption:= '∑';
        end;

        captions.count:= 18;
        with captions[0] do begin
          mergecols:= 1;
          textflags:= textflags + [tf_right];
          font:= tcolheaderfont.create;
          font.style:= font.style + [fs_bold];
        end;

        for j:= 1 to captions.count-1 do begin
          with captions[j] do begin
            font:= tcolheaderfont.create;
            font.height:= 11;
            font.xscale:= 0.7;
          end;
        end;

      end;
    end;
  end;

  grdForm18.appendrow(['82','28','13','99','13','21']);

end;
