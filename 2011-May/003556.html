tdatacols.selected is an array property:
http://www.freepascal.org/docs-html/ref/refsu32.html#x79-890006.4.3
the index type is gridcoordty.
"
uses
  mseeditglob;
[...]
  <yourgridinstance>.datacols.selected[<yourgridcoord>]:= 
<yourselectedvalue>;
  grid1.datacols.selected[makegridcoord(0,0)]:= true;
"
or
"
  grid1.datacols.selected[mgc(0,0)]:= true;
"
or
"
var
  co1: gridcoordty;
[...]
  co1.col:= 0;
  co1.row:= 0;
  grid1.datacols.selected[co1]:= true;
