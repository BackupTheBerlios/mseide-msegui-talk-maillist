Just for others to get the overall picture my code ended up like this:

procedure tmainfo.insertnewwidget(const sender: TObject);
var
  stringedit3: tstringedit;
begin
  stringedit3:=tstringedit.create(self);
  twidgetgrid1.insertwidget(stringedit3);
  stringedit3.gridvalue[y]:='blablabl';  //y is the selected row
end;
