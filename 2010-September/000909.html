Found a way to solve the problem, by using a small external program.

http://johnst.org/sw/exiftags/

Thanks to Eric Johnston for creating this little beauty.

(remember to add sysutils,process,classes to uses).


procedure tmainfo.openpicture(const sender: TObject);
var
  AProcess: TProcess;
  AStringList: TStringList;
begin
if tfiledialog1.execute(fdk_open) = mr_ok then
	begin
	  AStringList := TStringList.Create;
  	AProcess := TProcess.Create(nil);
  	AProcess.CommandLine := 'exiftags '+tfiledialog1.controller.filename;
  	AProcess.Options := AProcess.Options + [poWaitOnExit, poUsePipes];
  	AProcess.Execute;
  	AStringList.LoadFromStream(AProcess.Output);
  	tmemoedit1.text:=AStringList.text;
  	AStringList.Free;
  	AProcess.Free;
   timage1.bitmap.LoadFromfile(tfiledialog1.controller.filename);
  end;
end;
