//--------------------
program textreplace;

{$ifdef FPC}{$mode objfpc}{$h+}{$endif}
{$ifdef mswindows}{$apptype console}{$endif}

uses
  sysutils, msestrings;

var
   s_in,s_to: msestring;
   i: integer;

begin

  s_in:= 'Mother washed Dilbar, Dilbar and again Dilbar';
  s_to:= 'window frame';

  repeat
    i:= msestringsearch('Dilbar',s_in,0,[so_wholeword]);
    s_in:= replacetext(s_in,i,s_to);
  until i <= 0;

end.
//--------------------

Here, since 'window frame' is longer than 'Dilbar', it overwrites 6 chars  
after 'Dilbar' thus breaking word structure of the sentense.
How to keep the structure ?
