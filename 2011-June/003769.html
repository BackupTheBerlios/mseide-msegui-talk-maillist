procedure tmainfo.showdir(const sender: TObject);
var
  n:integer;
  SR: TSearchRec;
  extstrings:array of string;
begin
  tmemoedit1.text:='';
  setlength(extstrings,2);
  extstrings[0]:='.pas';
  extstrings[1]:='.mfm';
// just to fill some extensions for testing
  if FindFirst(tdirdropdownedit1.value+'*.*', faAnyFile - faDirectory,
SR) = 0 then
  repeat
    begin
      for n:=0 to 1 do
      if lowercase(ExtractFileExt(SR.name))=extstrings[n] then
	tmemoedit1.text:=tmemoedit1.text+SR.Name+chr(13)+chr(10);
    end;
  until FindNext(SR) <> 0;
  FindClose(SR);
end;
