As Martin says, use the MySQL tools inside an AProcess:

var
AProcess: TProcess;
hostname:string;
username:string;
userpass:string;
database:string;
tablename:string;

begin
.
.
AProcess := TProcess.Create(nil);
AProcess.CommandLine := 'mysqldump -h '+hostname+' -u '+username+' -p '
+password+' '+database+' '+tablename+' > tempfile.sql';
AProcess.Options := AProcess.Options + [poWaitOnExit, poUsePipes];
AProcess.Execute;
.
.
AProcess.CommandLine := 'mysql -h '+hostname+' -u '+username+' -p '
+password+' '+database+' < tempfile.sql';
AProcess.Options := AProcess.Options + [poWaitOnExit, poUsePipes];
AProcess.Execute;
.
.
end

remember to have process in your uses clause.
