Am Montag 15 August 2011, 16:19:47 schrieb Patrick Goupell:
> Is there an example of how to create / use my own dialog?
> 
What kind of dialog? A dialog normally is a form showed by 
twidget.show(ml_application) and returns a value. If the value type is 
msestring tdialogstringedit from tab 'Dialog' can be used.
Please create your form in onexecute event.
"
procedure tmainfo.exec(const sender: tcustomdataedit; var avalue: msestring;
               var modresult: modalresultty);
begin
 tmydialogfo.create(avalue,modresult);
end;
"

The dialogform:
"
type
 tmydialogfo = class(tmseform)
   ed: tstringedit;
   okbutton: tbutton;
  public
   constructor create(var avalue: msestring; var aresult: modalresultty);
 end;
implementation
uses
 dialog_mfm;

{ tmydialogfo }

constructor tmydialogfo.create(var avalue: msestring;
               var aresult: modalresultty);
begin
 inherited create(nil);
 ed.value:= avalue;
 aresult:= show(ml_application);
 avalue:= ed.value;
 release;
end;
"
 okbutton.modalresult is set to mr_ok.
